\section{Stany}

\begin{itemize}
  \item Izotermiczny - $T = $ const
  \item Izobaryczny - $p = $ const
  \item Izochoryczny - $V = $ const
  \item Adiabatyczny - $Q = 0$/const
\end{itemize}

$$
\partial W = -pdV
$$

\section{Gaz Doskonały}

$$
pV = nRT \Rightarrow V = \frac{RT}{p}
$$
gdzie $p$ to ciśnienie, $V$ to objętość, $n$ to ilość moli gazu, $R$
to stała gazowa, a $T$ to temperatura.

\subsection{Przykład}

Dla gazu $V = V(p, T)$, $dv = (\frac{\partial V}{\partial p})dp +
(\frac{\partial V}{\partial T})dt$.
$$
dv = -\frac{RT}{p^2}dp + \frac{R}{p}dT
$$
